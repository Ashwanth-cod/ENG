📝 Say / Output

say "Hello, world!"

📥 Input

let name be input("What is your name?")
say "Welcome " + name

🔢 Number Input

let age be number input("Your age:")
say "You are " + age + " years old"

🧠 Variables

let x be 5
let y be "Hello"
say x + 3
say y + " world"

🌀 Loops

repeat 5 times:
  say "Looping"
end
let i be 1
repeat while i is less than 4:
  say i
  let i be i + 1
end

🔀 Conditionals

let score be 85
if score is greater than 90:
  say "Excellent"
else if score is greater than 70:
  say "Good"
else:
  say "Try again"
end

⚙️ Functions

define greet(name):
  say "Hello " + name
end
call greet("Alice")

🧮 Math

let total be 5 + 3 * 2
say total

🗂 Comments

# This is a comment
say "ENG ignores comments"

🎯 Comparison

if x is equal to 5:
  say "x is 5"
end

🔚 Ending Programs

say "Finished"
stop
